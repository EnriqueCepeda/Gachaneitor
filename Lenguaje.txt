receta {
    nombre "Puré"
    descripcion "Esta es una receta mu rica que consiste en cocinarse un purecico puta madre. Que rico un 10"
    tiempo {
        total 20 min. # Comprobar con suma de tiempos parciales (Opcional)
        preparacion 5 min. # No hace falta comprobar, porque son verbos de persona
    }
    ingredientes {
        2 ud Calabaza
        5 ud Calabacín
        500 ml Agua
        500 gr Quesitos
        200 ml Leche
        1 kg Mantequilla
        Pimienta
        Sal
    }    
    pasos {
        - pelar calabacín, calabaza 
        - pelar calabaza
        - añadir agua 
        - añadir calabacín
        - añadir leche, mantequilla y pimienta
        - cocinar 25 minutos      
    }
}
